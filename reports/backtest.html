4. Visualizaes


















In[5]:






# Converter a coluna Date para datetime
df_results['Date'] = pd.to_datetime(df_results['Date'])

# Grfico 1: Preos das opes ao longo do tempo
plt.figure(figsize=(12, 8))

plt.subplot(2, 2, 1)
plt.plot(df_results['Date'], df_results['Stock_Price'], label='Preo da Ao', color='blue')
plt.plot(df_results['Date'], df_results['Call_Price'], label='Preo Call', color='green')
plt.plot(df_results['Date'], df_results['Put_Price'], label='Preo Put', color='red')
plt.title('Preos ao Longo do Tempo')
plt.xlabel('Data')
plt.ylabel('Preo (R$)')
plt.legend()
plt.xticks(rotation=45)

# Grfico 2: Delta das opes
plt.subplot(2, 2, 2)
plt.plot(df_results['Date'], df_results['Delta_Call'], label='Delta Call', color='green')
plt.plot(df_results['Date'], df_results['Delta_Put'], label='Delta Put', color='red')
plt.title('Delta das Opes')
plt.xlabel('Data')
plt.ylabel('Delta')
plt.legend()
plt.xticks(rotation=45)

# Grfico 3: Gamma e Vega
plt.subplot(2, 2, 3)
plt.plot(df_results['Date'], df_results['Gamma'], label='Gamma', color='purple')
plt.title('Gamma das Opes')
plt.xlabel('Data')
plt.ylabel('Gamma')
plt.legend()
plt.xticks(rotation=45)

# Grfico 4: Theta
plt.subplot(2, 2, 4)
plt.plot(df_results['Date'], df_results['Theta_Call'], label='Theta Call', color='orange')
plt.plot(df_results['Date'], df_results['Theta_Put'], label='Theta Put', color='brown')
plt.title('Theta das Opes')
plt.xlabel('Data')
plt.ylabel('Theta')
plt.legend()
plt.xticks(rotation=45)

plt.tight_layout()
plt.savefig('../reports/options_analysis.png', dpi=300, bbox_inches='tight')
plt.show()
